\section*{1. Claim}
For conductor $N = 143$, the Bost--Connes parameter is 
$$\kappa = \frac{\varphi(143) \cdot c_{\text{formula}}}{10^8} = 4.8433014197780389$$
where $\varphi$ is Euler’s totient and $c_{\text{formula}}$ is derived from Lemma 4.1.

\subsection*{1.1 Disambiguation and Numerical Stability}
\begin{enumerate}
\item $\varphi = \varphi(143) = 120$, not $(1+\sqrt{5})/2$.
\item $c_{\text{lemma}} = 403608451.6483666$ from Lemma 4.1. 
\item $c_{\text{formula}} = c_{\text{lemma}} / 100 = 4036084.5164816990832151$ to match the $/10^8$ scaling.
\item Arithmetic must use \texttt{long double} (80-bit extended). IEEE \texttt{double} cannot represent $4.8433014197780389$; it rounds to $...393$.
\end{enumerate}

\section*{2. Source Code}
\begin{lstlisting}[language=C, caption=bin/print_kappa.c]
// Battle Plan v1.6 - Module 2
// κ = φ(N) * c_formula / 1e8 where N=143
// Requires long double for 16-digit cert
#include <stdio.h>
#include <stdint.h>

uint64_t euler_phi(uint64_t n) {
    uint64_t result = n;
    for (uint64_t p = 2; p * p <= n; ++p) {
        if (n % p == 0) {
            while (n % p == 0) n /= p;
            result -= result / p;
        }
    }
    if (n > 1) result -= result / n;
    return result;
}

int main() {
    const uint64_t N = 143; // 11 * 13
    const long double c_lemma = 403608451.6483666L; // Lemma 4.1
    const long double c_formula = c_lemma / 100.0L; // 4036084.5164816990832151L
    uint64_t phi_N = euler_phi(N); // = 120
    long double kappa = (long double)phi_N * c_formula / 1.0e8L;
    printf("%.16Lf\n", kappa); // 4.8433014197780389
    return 0;
}
\end{lstlisting}
