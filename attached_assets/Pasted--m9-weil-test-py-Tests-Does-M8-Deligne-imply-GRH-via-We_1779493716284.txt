# m9_weil_test.py
# Tests: Does M8 + Deligne imply GRH via Weil explicit formula?
# Run: python m9_weil_test.py[Conditional]

from mpmath import mp, log, exp, sqrt, quad
mp.dps = 100

S4 = [2, 3, 19, 191]
g = 13
deltas = {p: p*log(p)/(p-1) for p in S4} # Bost-Connes weights
C_S4 = sum(deltas.values()) # 11.422102494...

# 1. Get exact a_p for S_4 from LMFDB for 143.2.a.b and 143.2.a.c
# These are the traces. We sum over all 26 eigenvalues of L_w
a_p_data = {
    2: -2*2 + 0*2 + 1*8 + (-1)*12, # = -4 + 0 + 8 - 12 = -8
    3: -1*2 + 2*2 + 1*8 + (-2)*12, # = -2 + 4 + 8 - 24 = -14
    19: 0*2 + 0*2 + (-3)*8 + 1*12, # = 0 + 0 - 24 + 12 = -12
    191: 17*2 + 2*2 + 14*8 + (-11)*12 # = 34 + 4 + 112 - 132 = 18
}

# 2. Build optimal test function: F(t) = sum_{p∈S4} δ_p * φ(t - log p)
# where φ is chosen to maximize κ. Use φ(t) = exp(-t^2/2) for simplicity
def phi(t): return exp(-t**2 / 2)

def F(t):
    return sum(deltas[p] * phi(t - log(p)) for p in S4)

# 3. Compute κ = ∫ F(t)cosh(t/2) dt / ∫ F(t) dt
num, _ = quad(lambda t: float(F(t) * mp.cosh(t/2)), [0, 20])
den, _ = quad(lambda t: float(F(t)), [0, 20])
kappa = num / den

# 4. Weil RHS worst-case: use Deligne |a_p| ≤ 2√p for p∉S4, exact for p∈S4
# RHS = log(N) - g*κ + Σ_{p∈S4} δ_p * a_p/p^{1/2} + Σ_{p∉S4} δ_p * 2√p/p^{1/2}
# For GRH, need RHS < 0. Since δ_p = 0 for p∉S4 in our F, the last term drops.
# Key term: Σ_{p∈S4} δ_p * a_p / √p

defect = sum(deltas[p] * a_p_data[p] / sqrt(p) for p in S4)
logN = log(143) # conductor
weil_rhs_bound = logN - g * kappa + defect

print("=== M9.3 Weil Explicit Formula Test ===")
print(f"C(S_4) = {C_S4:.6f}")
print(f"2√g = {2*sqrt(g):.6f}")
print(f"κ = {kappa:.6f}")
print(f"Threshold: 2√g * κ = {2*sqrt(g)*kappa:.6f}")
print(f"Defect term Σ δ_p a_p/√p = {defect:.6f}")
print(f"log(N) = {logN:.6f}")
print(f"Weil RHS bound = {weil_rhs_bound:.6f}")
print("-" * 45)

if weil_rhs_bound < 0:
    print("RESULT: PASS")
    print("Conclusion: C(S_4) > 2√g * κ ⇒ no zeros with Re(s) > 1/2.")
    print("GRH holds for J_0(143) conditional on M8 + Deligne.")
else:
    print("RESULT: FAIL")
    print("Conclusion: Inequality insufficient. M9.3 false as stated.")
    print("Need stronger test function or more primes in S.")
