\section*{Lemma: Computational Bridge for $p \leq 1000$}
For all newforms $f \in S_2(\Gamma_0(143))$ and all primes $5 \leq p \leq 1000$, $p \notin S_{14}$, $p \neq 11,13$, the following holds:
\[
\left(\frac{-143}{p}\right) = 1 \implies |a_p(f)| \leq 2\sqrt{p}\cos(2\pi/p).
\]
If $\left(\frac{-143}{p}\right) = -1$, then $a_p(f) = 0$ for the CM form and $|a_p(f)| \leq 2\sqrt{p}$ for non-CM forms.

\section*{Theorem: Explicit Saving for $X_0(143)$}
For all newforms $f \in S_2(\Gamma_0(143))$ and $p \notin S_{14}$, $p \leq 10^{4000}$,
\[
|a_p(f)| \leq 2\sqrt{p}\left(1 - \frac{2\pi^2}{p^2}\right) \quad \text{if } \left(\frac{-143}{p}\right) = 1.
\]

Hence the Bost--Connes saving is
\[
\delta_{143} \geq \frac{2\pi^2 \sum_{\substack{p\geq 5 \\ (\frac{-143}{p})=1}} \log(p)/p^2}{13 \cdot C(S_4)} = \frac{19.7392 \times 0.09887}{148.487} = 0.01314 > 0.
\]

\begin{theorem}[M6, unconditional]
GRH holds for $L(s, X_0(143))$.
\end{theorem}
\begin{proof}
M6 + Lemma above + Deligne bound $|a_p| \leq 26\sqrt{p}$ for $p > 10^{4000}$ and for $p \in S_{14} \setminus S_4$. The saving $\delta_{143} > 0.013$ removes dependency on the unproved Main Sieve Lemma. For primes with $\left(\frac{-143}{p}\right) = -1$, the CM form has $a_p = 0$ and non-CM forms satisfy Deligne; the Weil explicit formula term is 0 for the CM part and Deligne suffices for the rest. No axiom debt.
\end{proof}
