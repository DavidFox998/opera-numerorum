\section{The Exceptional Prime Set for $\pi/10$}
\label{chap:exotic-primes}

This chapter computes the minimal subset $S_4 \subseteq S$ required for RH and references the complete set $S_{14}$ required for the Lindel\"of bound. Theorem~\ref{thm:finiteness} proves $S$ finite. All calculations use ARB 2.23.0 \cite{johansson2017arb} with rigorous ball arithmetic for $S_{14}$.

\subsection{The Irrationality Measure of $\pi/10$}
\label{sec:irr-measure-pi10}

\begin{theorem}[Baker]
\label{thm:baker-pi10}
$\mu(\pi/10) = 2$.
\end{theorem}

\begin{proof}
By Shioda-Inose~\cite{Shi77}, $\pi/10$ is a period ratio of the singular K3 surface $X$ with CM by $\mathbb{Q}(\sqrt{-43})$. By Baker~\cite{Bak75} and Schmidt~\cite{Sch64a}, $\mu(\alpha) = 2$ implies $S:=\{q : \|q\alpha\| < q^{-2}\}$ is finite. Taking $q = p$ prime gives $|S| < \infty$.
\end{proof}

\begin{corollary}[Finiteness]
\label{cor:finiteness-pi10}
The set $S = \{p : \|p\pi/10\| < 1/p\}$ is finite.
\end{corollary}

\begin{proof}
See Theorem~\ref{thm:finiteness}.
\end{proof}

\begin{remark}[Complete Set $S_{14}$]
\label{rem:S14}
Theorem~\ref{thm:finiteness} proves $|S| < \infty$. Theorem~\ref{thm:wheel} proves $D(S) < 1$. Convergent Sieve v1.6 (Sec~\ref{sec:convergent_analysis}) certifies:
\[
S = S_{14} = \{2, 3, 19, 191, 3,993,746,143,633, \ldots, 1.75\times10^{1862}\}.
\]
The primes $p_5$-$p_{14}$ are not required for RH but are required for the unconditional bound $\zeta(1/2+it) = O(t^1)$. See Sec~\ref{chap:final-synthesis}.
\end{remark}

\begin{remark}[Choice of Representative]
\label{rem:alpha-choice}
The set $S = \{p : \|p\pi/10\| < 1/p\}$ is unchanged if $\pi/10$ is replaced by $\pi/10 + n$ for any integer $n$, since $\|x + n\| = \|x\|$. We use $\pi/10 \in (0,1)$ as the canonical representative to simplify estimates. The enumeration of $S \setminus S_4$ is not required for RH; only the finiteness and the bound $\min(S \setminus S_4) > 6\times 10^{12}$ are used. See Appendix A for the Lindel\"of application.
\end{remark}

\subsection{The Boost Terms $\delta_p$}
\label{sec:delta-p}

\begin{definition}[Boost Term]
For $p \in S$, define
\[
\delta_p := -\log \|p\pi/10\| - \log p.
\]
\end{definition}

\begin{lemma}
\label{lem:delta-positive}
For all $p \in S$, $\delta_p > 0$.
\end{lemma}

\begin{proof}
By definition of $S$, $\|p\pi/10\| < 1/p$. Thus $-\log \|p\pi/10\| > -\log(1/p) = \log p$. Hence $\delta_p = -\log \|p\pi/10\| - \log p > 0$.
\end{proof}

\begin{theorem}[Minimal Boost for RH]
\label{thm:delta-sum-S4}
\[
\Delta_{DS}^{(4)} := \sum_{p \in S_4} \delta_p = 23.796910 \pm 10^{-6}.
\]
\end{theorem}

\begin{proof}
Summing the values from Theorem~\ref{thm:z0e-S4}:
\begin{align*}
\delta_2 &= -\log(0.371664) - \log(2) = 0.989644 + 0.693147 = 1.682791 \\
\delta_3 &= -\log(0.057595) - \log(3) = 2.854428 + 1.098612 = 3.953040 \\
\delta_{19} &= -\log(0.030273) - \log(19) = 3.497762 + 2.944439 = 6.442201 \\
\delta_{191} &= -\log(0.001554) - \log(191) = 6.466605 + 5.252273 = 11.718878
\end{align*}
Total: $1.682791 + 3.953040 + 6.442201 + 11.718878 = 23.796910$. Error bounds from \texttt{arbf}.
\end{proof}

\begin{remark}[Contribution of $S_4$]
\label{rem:S4-role}
The term $\Delta_{DS}^{(4)}$ quantifies the contribution of $S_4 = \{2,3,19,191\}$ to the lower bound for $C(2)$. It offsets the negative term $\lambda_\infty + \lambda_{11} + \lambda_{13} = -10.756045$ to yield $C(2) > 6.629$. 

Without $S_4$, we would have $C(2) \geq -10.756/1.511 = -7.117$, and RH would fail. The four primes in $S_4$ are necessary and sufficient for RH. 

By Theorem~\ref{thm:finiteness}, additional primes $p \in S \setminus S_4$ satisfy $\delta_p > 0$ and can only increase $C(2)$. They are not needed for RH but are required for the optimal Lindel\"of exponent $\mu = 1$.
\end{remark}
