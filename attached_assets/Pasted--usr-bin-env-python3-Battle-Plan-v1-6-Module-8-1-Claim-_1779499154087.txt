#!/usr/bin/env python3
# Battle Plan v1.6 - Module 8.1
# Claim: Hecke traces a_p for S_2(Gamma_0(143)), 5 <= p <= 1000
# Standard: One claim, one PDF, one SHA. No external data without hash.
# Dependencies: M8 SHA e2d70821cd66588cd715dfe37a44122130f88d15584738f5f64a02ff7f7b0002

import json, hashlib, csv, sys, subprocess, urllib.request
from mpmath import mp

mp.dps = 50
OUT_CSV = "143_traces.csv"
OUT_LOG = "lmfdb_api.log"

# M8 certified structure: label, multiplicity, LMFDB URL stub
FORMS = [
    ('11.2.a.a', 2, 'mf/11/2/a/a/'), # oldform from level 11
    ('143.2.a.a', 1, 'mf/143/2/a/a/'), # newform, rational
    ('143.2.a.b', 4, 'mf/143/2/a/b/'), # newform, dim 4
    ('143.2.a.c', 6, 'mf/143/2/a/c/'), # newform, dim 6
]

def sha256_bytes(b: bytes) -> str:
    return hashlib.sha256(b).hexdigest()

def fetch_lmfdb_json(url: str) -> bytes:
    """Fetch LMFDB API, log raw bytes, return them. Hash is bound."""
    with urllib.request.urlopen(url) as resp:
        data = resp.read()
    with open(OUT_LOG, 'ab') as log:
        log.write(b"URL: " + url.encode() + b"\n")
        log.write(b"SHA-256: " + sha256_bytes(data).encode() + b"\n")
        log.write(b"BYTES: " + str(len(data)).encode() + b"\n\n")
    return data

def parse_ap_from_json(raw: bytes, p: int) -> int:
    """Extract a_p from LMFDB JSON. Crash if missing."""
    j = json.loads(raw.decode())
    # LMFDB: data['hecke_n'] is list where index p gives a_p
    # Different for rational vs algebraic forms
    if 'hecke_n' in j:
        return int(j['hecke_n'][p])
    if 'traces' in j: # rational case
        return int(j['traces'][p])
    raise ValueError(f"Cannot parse a_{p} from JSON")

def main():
    primes = [p for p in range(5, 1001) if mp.isprime(p)]

    # Pre-fetch all JSON and hash it
    cache = {}
    for label, mult, stub in FORMS:
        url = f"https://www.lmfdb.org/api/{stub}"
        raw = fetch_lmfdb_json(url)
        cache[label] = raw
        print(f"Fetched {label}: {len(raw)} bytes, SHA {sha256_bytes(raw)}", file=sys.stderr)

    rows = []
    for p in primes:
        row = {'p': p}
        for label, mult, _ in FORMS:
            ap = parse_ap_from_json(cache[label], p)
            row[label] = ap
        rows.append(row)

    # Write CSV
    with open(OUT_CSV, 'w', newline='') as f:
        writer = csv.DictWriter(f, fieldnames=['p'] + [l for l,_,_ in FORMS])
        writer.writeheader()
        writer.writerows(rows)

    # Final SHA binding
    with open(OUT_CSV, 'rb') as f:
        csv_sha = sha256_bytes(f.read())
    with open(OUT_LOG, 'rb') as f:
        log_sha = sha256_bytes(f.read())

    print(f"143_traces.csv SHA-256: {csv_sha}")
    print(f"lmfdb_api.log SHA-256: {log_sha}")
    print("M8.1 STATUS: CERTIFIED if SHAs recorded in PDF", file=sys.stderr)

if __name__ == "__main__":
    main()
